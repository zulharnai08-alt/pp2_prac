import re
import json

# читаем файл | read receipt file
with open("raw.txt", "r", encoding="utf-8") as file:
    text = file.read()

lines = text.split("\n")

products = []   # список товаров | list of products
prices = []     # список цен | list of prices

# regex для цены | regex for price
price_pattern = re.compile(r'\d[\d\s]*,\d{2}')

for i, line in enumerate(lines):

    # ищем цену | find price
    price_match = price_pattern.fullmatch(line.strip())

    if price_match:

        price = price_match.group().replace(" ", "").replace(",", ".")
        prices.append(float(price))

        # предыдущая строка = товар | previous line is product
        product = lines[i-2].strip()

        if product and not product.isdigit():
            products.append(product)

# считаем сумму | calculate total
total = sum(prices)

# ищем дату и время | find date and time
datetime_match = re.search(r'\d{2}\.\d{2}\.\d{4}\s+\d{2}:\d{2}:\d{2}', text)

date = None
time = None

if datetime_match:
    date, time = datetime_match.group().split()

# ищем способ оплаты | find payment method
payment_match = re.search(r'(Банковская карта|Cash|Card)', text)

payment_method = payment_match.group() if payment_match else None

# структурированный результат | structured result
result = {
    "products": products,
    "prices": prices,
    "total": total,
    "date": date,
    "time": time,
    "payment_method": payment_method
}

# вывод JSON | print JSON
print(json.dumps(result, indent=4, ensure_ascii=False))
